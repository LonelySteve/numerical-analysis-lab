\subsubsection{问题一}

\begin{breakablealgorithm}
    \caption{高斯顺序消去法}
    \begin{algorithmic}[1]
        \REQUIRE{矩阵 $A$ 的对角线元素均不为 0}
        \STATE $B \leftarrow [A, b]$ \COMMENT{通过合并系数矩阵 $A$ 和常数矩阵 $b$ 获得增广矩阵 $B$ }
        \STATE $[n, m] \leftarrow size(B)$ \COMMENT{求增广矩阵 B 的行数和列数，分别记做 n 和 m}
        \FOR{$k=1$ \TO $n$}
        \FOR{$o=k+1$ \TO $n$}
        \FOR{$p=0$ \TO $m$}
        \STATE $B_{o, p} = B_{o, p} - (B_{o, k} / B_{k, k}) * B_{k, p}$
        \ENDFOR
        \ENDFOR
        \ENDFOR
        \STATE $x \leftarrow zeros(m-1, 1)$ \COMMENT{分配 x 的空间}
        \FOR{$k=n$ \TO $1$ by step $-1$}
        \FOR{$q=k+1$ \TO $m-1$}
        \STATE $x_k \leftarrow (B_{k, m} - B_{k, q} * x_q ) / B_{k, k}$
        \ENDFOR
        \ENDFOR
    \end{algorithmic}
\end{breakablealgorithm}

\subsubsection{问题二}

\begin{breakablealgorithm}
    \caption{cholesky 算法}
    \begin{algorithmic}[1]
        \REQUIRE{被分解的矩阵必须为厄米特矩阵或符合正定矩阵定义的矩阵}
        \STATE $[n, \sim] \leftarrow size(A)$ \COMMENT{获取矩阵 A 的行数 n}
        \STATE $L \leftarrow zeros(n, n)$ \COMMENT{分配矩阵 L 的空间}
        \FOR{$j=1$ \TO $n$}
        \FOR{$i=j$ \TO $n$}
        \IF{$i == j$}
        \STATE $L_{i, j} \leftarrow \sqrt{A_{i, j}} - \sum_{k=1}^{j-1}L_{j, k}^2$
        \ELSE
        \STATE $L_{i, j} \leftarrow (A_{i, j} - \sum_{k=1}^{j-1}L_{i, k}L_{j, k}) / L_{j, j}$
        \ENDIF
        \ENDFOR
        \ENDFOR
    \end{algorithmic}
\end{breakablealgorithm}

\subsubsection{问题三}

\begin{breakablealgorithm}
    \caption{追赶法}
    \begin{algorithmic}[1]
        % \REQUIRE{系数矩阵 A 必须为三对角线矩阵}
        \STATE $[n, \sim] \leftarrow size(A)$ \COMMENT{获取矩阵 A 的行数 n}
        \STATE $a \leftarrow ones(n - 1, 1)$  \COMMENT{初始化 a 向量的空间}
        \STATE $b \leftarrow ones(n, 1)$ \COMMENT{初始化 b 向量的空间}
        \STATE $c \leftarrow ones(n - 1, 1)$\COMMENT{初始化 c 向量的空间}
        \COMMENT{构造 a，b，c 向量}
        \FOR{$i=1$ \TO $n - 1$}
        \STATE $b_{i, 1} \leftarrow A_{i, i}$
        \STATE $a_{i, 1} \leftarrow A_{i + 1, i}$
        \STATE $c_{i, 1} \leftarrow A_{i, i + 1}$
        \ENDFOR
        \STATE $b_{n, 1} \leftarrow A_{n, n}$
        \STATE $d \leftarrow zeros(n - 1, 1)$ \COMMENT{初始化 d 向量（$\beta$ 向量）的空间}
        \STATE $d_1 \leftarrow c_1 / b_1$
        \COMMENT{构造 d 向量}
        \FOR{$i=2$ \TO $n - 1$}
        \STATE $d_i \leftarrow c_i / (b_i - a_{i - 1} * d_{i - 1})$
        \ENDFOR
        \STATE $y \leftarrow zeros(n, 1)$ \COMMENT{初始化 y 向量的空间}
        \STATE $y_1 \leftarrow f_1 / b_1$
        \FOR{$i=2$ \TO $n$}
        \STATE $y_i \leftarrow (f_i - a_{i - 1} * y_{i - 1}) / ((b_i - a_{i - 1}) * d_{i - 1})$
        \STATE $x \leftarrow zeros(n, 1)$ \COMMENT{初始化解 x 向量的空间}
        \STATE $x_n \leftarrow y_n$
        \ENDFOR
        \FOR{$i=n-1$ \TO $1$ by step $-1$}
        \STATE $x_i = y_i - d_i * x_{i + 1}$
        \ENDFOR
    \end{algorithmic}
\end{breakablealgorithm}
